1. Safely access nested properties that might not exist.

2. Without optional chaining :

            let user = {
            name: "John",
            address: {
                city: "New York"
            }
            };

    // This is safe:
    console.log(user.address.city);  // "New York"

    // But if address doesn't exist:
    let user2 = { name: "Sarah" };

    // console.log(user2.address.city);  
    ❌ ERROR! Cannot read property 'city' of undefined

3. With Optional chaining :

    let user2 = { name: "Sarah" };
    console.log(user2.address?.city);  // undefined (no error!)

4. EXAMPLES 

    let user = {
        name: "John",
        getAddress: function() {
            return { city: "NY" };
        }
    };

// Safe method call
console.log(user.getAddress?.());  // {city: "NY"}

let user2 = { name: "Sarah" };
console.log(user2.getAddress?.());  // undefined (no error)

// Array access
let users = [{ name: "John" }];
console.log(users[0]?.name);  // "John"
console.log(users[5]?.name);  // undefined (no error)
